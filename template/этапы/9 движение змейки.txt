/**
 * Получить рандомное целое число в диапозоне от min до max
 * @param min
 * @param max
 * @returns {number}
 */
function randomInteger(min, max) {
    let rand = min - 0.5 + Math.random() * (max - min + 1);
    return Math.round(rand);
}
